       ```go
       import (
           "time"
           "gorm.io/gorm"
       )

       type Invitation struct {
           ID         int64          `gorm:"primaryKey;autoIncrement" json:"id"`
           OrgID      int64          `gorm:"column:org_id;index;not null" json:"org_id"`
           Token      string         `gorm:"column:token;type:varchar(64);uniqueIndex;not null" json:"token"`
           Position   string         `gorm:"column:position;type:varchar(100)" json:"position"`
           Status     string         `gorm:"column:status;type:varchar(20);not null;default:pending" json:"status"` // pending/used/expired/cancelled
           CreatedBy  int64          `gorm:"column:created_by;not null" json:"created_by"`
           CreatedAt  time.Time      `gorm:"column:created_at;autoCreateTime" json:"created_at"`
           ExpiresAt  time.Time      `gorm:"column:expires_at;not null" json:"expires_at"`
           UsedAt     *time.Time     `gorm:"column:used_at" json:"used_at"`
           EmployeeID *int64         `gorm:"column:employee_id" json:"employee_id"`
       }

       func (Invitation) TableName() string {
           return "invitations"
       }
       ```
       注意: Invitation 不嵌入 BaseModel，使用独立的简化字段（id, org_id, created_by, created_at — 不需要 updated_by/updated_at/deleted_at，邀请创建后仅更新 status/used_at/employee_id）。

       ```go
       type InvitationService struct {
           invRepo     *InvitationRepository
           empRepo     *Repository           // 复用 Plan 01 的 Employee Repository
           cryptoCfg   config.CryptoConfig
       }
       func NewInvitationService(invRepo *InvitationRepository, empRepo *Repository, cryptoCfg config.CryptoConfig) *InvitationService
       ```

       关键方法:
       - `CreateInvitation(ctx, orgID, userID int64, req *CreateInvitationRequest) (*CreateInvitationResponse, error)`:
         1. `crypto_rand.Read(token)` 生成 32 字节 → `hex.EncodeToString` → 64 字符 token（per D-01）
         2. `ExpiresAt = time.Now().Add(7 * 24 * time.Hour)` (per D-04)
         3. invRepo.Create 保存
         4. 返回 CreateInvitationResponse{Token, InviteURL: "/invite/"+token, ExpiresAt}

       - `GetInvitationDetail(ctx, token string) (*InvitationDetailResponse, error)`:
         1. invRepo.FindByToken(token)
         2. 检查 status 不是 pending → 返回 "邀请已使用/已取消" 错误
         3. 检查 time.Now().After(expiresAt) → 返回 "邀请已过期" 错误
         4. 查询 org_name
         5. 返回 InvitationDetailResponse（不含敏感信息）

       - `SubmitInvitation(ctx, token string, req *SubmitInvitationRequest) error`:
         1. invRepo.FindByToken(token)
         2. 验证 status == pending，否则返回 "邀请已使用/已取消"
         3. 验证 time.Now().Before(expiresAt)，否则返回 "邀请已过期"
         4. 检查手机号唯一性: empRepo.FindByPhoneHash(orgID, hash) → 如存在返回 "该手机号已存在"
         5. 检查身份证号唯一性: empRepo.FindByIDCardHash(orgID, hash)
         6. 从身份证提取 gender + birthDate（复用 extractFromIDCard）
         7. 加密敏感字段，生成 hash
         8. 创建 Employee{Status: "pending", CreatedBy: inv.CreatedBy} (per D-03, D-05)
         9. 更新 Invitation{Status: "used", EmployeeID: emp.ID, UsedAt: now}
         10. 以上操作在事务中执行（invRepo.db.Transaction）

       - `ListInvitations(ctx, orgID int64, query ListInvitationsQuery) ([]InvitationListItem, int64, error)`: 分页列表
       - `CancelInvitation(ctx, orgID, id int64) error`: 仅 status=pending 可取消

       - `ConfirmOnboarding(ctx, orgID, employeeID int64) error`:
         1. empRepo.FindByID(orgID, employeeID)
         2. 验证 status == "pending"
         3. empRepo.Update(orgID, employeeID, map[string]interface{}{"status": "active"})

       ```go
       func (h *InvitationHandler) RegisterRoutes(rg *gin.RouterGroup, authMiddleware gin.HandlerFunc) {
           authGroup := rg.Group("")
           authGroup.Use(authMiddleware)

           // 公开接口（无需认证）— 员工填写信息
           rg.GET("/invitations/:token", h.GetInvitationDetail)
           rg.POST("/invitations/:token/submit", h.SubmitInvitation)

           // 需要认证的接口 — 老板操作
           authGroup.POST("/invitations", middleware.RequireRole("owner", "admin"), h.CreateInvitation)
           authGroup.GET("/invitations", h.ListInvitations)
           authGroup.DELETE("/invitations/:id", middleware.RequireRole("owner", "admin"), h.CancelInvitation)
           authGroup.POST("/employees/:id/confirm", middleware.RequireRole("owner", "admin"), h.ConfirmOnboarding)
       }
       ```

       Handler 方法:
       - CreateInvitation: ShouldBindJSON → c.GetInt64("org_id") + c.GetInt64("user_id") → svc.CreateInvitation → Success
       - ListInvitations: ShouldBindQuery → svc.ListInvitations → PageSuccess
       - GetInvitationDetail: c.Param("token") → svc.GetInvitationDetail → Success
       - SubmitInvitation: c.Param("token") + ShouldBindJSON → svc.SubmitInvitation → Success(gin.H{"message": "提交成功"})
       - CancelInvitation: strconv.ParseInt(c.Param("id")) → svc.CancelInvitation → Success
