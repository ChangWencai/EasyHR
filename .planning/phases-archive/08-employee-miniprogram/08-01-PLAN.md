    package wxmp

    // Auth
    type LoginRequest struct {
        Phone string `json:"phone" binding:"required,phone"`
        Code  string `json:"code" binding:"required,len=6"`
    }
    type LoginResponse struct {
        AccessToken  string `json:"access_token"`
        RefreshToken string `json:"refresh_token"`
        ExpiresIn   int    `json:"expires_in"`
        Member      MemberInfo `json:"member"`
    }
    type MemberInfo struct {
        UserID      uint   `json:"user_id"`
        EmployeeID  uint   `json:"employee_id"`
        OrgID       uint   `json:"org_id"`
        Name        string `json:"name"`
        Phone       string `json:"phone"`
        Role        string `json:"role"`
        HasWechat   bool   `json:"has_wechat"`
    }

    // Payslip
    type PayslipSummary struct {
        ID          uint   `json:"id"`
        Year        int    `json:"year"`
        Month       int    `json:"month"`
        GrossPay    string `json:"gross_pay"`
        NetPay      string `json:"net_pay"`
        Status      string `json:"status"` // pending/confirmed/paid
        SignedAt    string `json:"signed_at,omitempty"`
        PaidAt      string `json:"paid_at,omitempty"`
    }
    type PayslipDetail struct {
        ID            uint   `json:"id"`
        Year          int    `json:"year"`
        Month         int    `json:"month"`
        GrossPay      string `json:"gross_pay"`
        NetPay        string `json:"net_pay"`
        Items         []PayrollItemDTO `json:"items"`
        SocialDeduct  string `json:"social_deduct"`
        TaxDeduct     string `json:"tax_deduct"`
        OtherDeduct   string `json:"other_deduct"`
        SignedAt      string `json:"signed_at,omitempty"`
        PaidAt        string `json:"paid_at,omitempty"`
    }
    type PayrollItemDTO struct {
        Name   string `json:"name"`
        Type   string `json:"type"` // earning/deduction
        Amount string `json:"amount"`
    }

    // Contract
    type ContractDTO struct {
        ID           uint   `json:"id"`
        ContractType string `json:"contract_type"` // labor/internship
        Status       string `json:"status"` // pending/signed/expired
        StartDate    string `json:"start_date"`
        EndDate      string `json:"end_date"`
        SignedAt     string `json:"signed_at,omitempty"`
        PDFURL       string `json:"pdf_url,omitempty"`
    }

    // Social Insurance
    type SocialInsuranceDTO struct {
        PaymentMonth   string `json:"payment_month"`
        City           string `json:"city"`
        BaseAmount     string `json:"base_amount"`
        Pension        string `json:"pension"`
        Medical        string `json:"medical"`
        Unemployment   string `json:"unemployment"`
        TotalPersonal  string `json:"total_personal"` // sum of above (NO employer amount)
    }

    // Expense
    type ExpenseRequest struct {
        Type        string   `json:"type" binding:"required,oneof=travel traffic entertainment office other"`
        Amount      string   `json:"amount" binding:"required"`
        Description string   `json:"description"`
        Attachments []string `json:"attachments"` // OSS file keys, max 9
    }
    type ExpenseDTO struct {
        ID          uint   `json:"id"`
        Type        string `json:"type"`
        Amount      string `json:"amount"`
        Description string `json:"description"`
        Status      string `json:"status"` // pending/approved/rejected/paid
        Attachments []string `json:"attachments"`
        CreatedAt   string `json:"created_at"`
        ApprovedAt   string `json:"approved_at,omitempty"`
        RejectedAt   string `json:"rejected_at,omitempty"`
        RejectReason string `json:"reject_reason,omitempty"`
        PaidAt       string `json:"paid_at,omitempty"`
        PaidMethod   string `json:"paid_method,omitempty"`
    }

    // SMS Verification
    type VerifyPayslipRequest struct {
        Code string `json:"code" binding:"required,len=6"`
    }
    type VerifyPayslipResponse struct {
        VerifyToken string `json:"verify_token"`
        ExpiresAt  string `json:"expires_at"`
    }

    // OSS
    type OssUploadURLRequest struct {
        FileName string `json:"file_name" binding:"required"`
        FileSize int64  `json:"file_size" binding:"required,gt=0"`
    }
    type OssUploadURLResponse struct {
        UploadURL string `json:"upload_url"`
        FileKey   string `json:"file_key"`
    }

    **middleware.go** — WXMPMemberAuth:
    ```go
    func WXMPMemberAuth(jwtSecret []byte) gin.HandlerFunc
    ```
    - Parse Bearer token from Authorization header.
    - Validate JWT; extract user_id, employee_id, org_id, role from claims.
    - Reject if role != "MEMBER" (all WXMP endpoints are MEMBER-only).
    - Set c.Set("user_id", userID), c.Set("employee_id", employeeID), c.Set("org_id", orgID).
    - Return 401 for invalid token, 403 for non-MEMBER role.

    func RegisterWXMPRouter(rg *gin.RouterGroup, db *gorm.DB, jwtSecret []byte, smsSvc *sms.Service) {
        repo := NewRepository(db)
        svc := NewWXMPService(repo, jwtSecret, smsSvc)
        handler := NewHandler(svc)
        rg := rg.Group("/wxmp")
        // Auth routes (no auth required)
        auth := rg.Group("/auth")
        auth.POST("/login", handler.Login)
        auth.POST("/wechat/bind", handler.BindWechat)
        // Member routes (MEMBER auth required)
        member := rg.Group("")
        member.Use(WXMPMemberAuth(jwtSecret))
        member.GET("/payslips", handler.ListPayslips)
        member.POST("/payslips/:id/verify", handler.VerifyPayslip)
        member.GET("/payslips/:id", handler.GetPayslipDetail)

        member.GET("/contracts/:id/pdf", handler.GetContractPDF)
        member.GET("/social-insurance", handler.GetSocialInsurance)
        member.POST("/expenses", handler.CreateExpense)
        member.GET("/expenses", handler.ListExpenses)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Untrusted internet → /api/v1/wxmp/* | All WXMP endpoints; JWT required except /auth/login |
| Mini Program → /wxmp/payslips/:id | Sensitive payslip detail; SMS verify_token required in addition to JWT |
| Employee data isolation | user_id from JWT claims only — no request param can override |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-01 | Spoofing | /api/v1/wxmp/auth/login | mitigate | SMS code verification before issuing JWT; code is single-use and expires in 5min |
| T-08-02 | Spoofing | WXMP MEMBER token | mitigate | JWT HMAC signed; MEMBER role enforced server-side on every route via WXMPMemberAuth middleware |
| T-08-03 | Tampering | payslip_id request param | mitigate | employee_id always from JWT claims, not URL param; verify that payslip belongs to the employee before returning detail |
| T-08-04 | Tampering | employee_id in request | mitigate | All queries use employee_id from JWT c.Get("employee_id"); request params for IDs verified against ownership |
| T-08-05 | Information Disclosure | Payslip detail | mitigate | SMS verify_token required for payslip detail; verify_token is short-lived (15min) and single-payslip |
| T-08-06 | Information Disclosure | Other employee data | mitigate | All repository queries filter by user_id→employee_id→org_id chain from JWT; tenant isolation via orgID |
| T-08-07 | Denial of Service | SMS verify flood | mitigate | Rate limit SMS sends: max 5 per phone per hour (reuse Phase 1 rate limit config) |
| T-08-08 | Repudiation | Payslip sign | mitigate | signed_at timestamp recorded server-side; non-repudiable |
</threat_model>
