    func (h *Handler) RegisterRoutes(rg *gin.RouterGroup, authMiddleware gin.HandlerFunc):
    注册路由 (per D-26 RBAC):
    ```
    authGroup := rg.Group("").Use(authMiddleware)

    // 税率表 — 所有角色可查询
    authGroup.GET("/tax/brackets", h.ListTaxBrackets)
    authGroup.POST("/tax/brackets/seed", middleware.RequireRole("owner"), h.SeedBrackets)

    // 专项附加扣除 — OWNER/ADMIN 管理，MEMBER 只读
    authGroup.POST("/tax/deductions", middleware.RequireRole("owner", "admin"), h.CreateDeduction)
    authGroup.GET("/tax/deductions", h.ListDeductions)
    authGroup.GET("/tax/deductions/:id", h.GetDeduction)
    authGroup.PUT("/tax/deductions/:id", middleware.RequireRole("owner", "admin"), h.UpdateDeduction)
    authGroup.DELETE("/tax/deductions/:id", middleware.RequireRole("owner", "admin"), h.DeleteDeduction)

    // 个税计算
    authGroup.POST("/tax/calculate", middleware.RequireRole("owner", "admin"), h.CalculateTax) — 传入 {employee_id, year, month, gross_income}
    authGroup.POST("/tax/calculate-contract", middleware.RequireRole("owner", "admin"), h.CalculateTaxFromContract) — 传入 {employee_id, year, month}

    // 个税记录查询
    authGroup.GET("/tax/records", h.ListTaxRecords)
    authGroup.GET("/tax/records/:id", h.GetTaxRecord)
    authGroup.GET("/tax/my-records", h.GetMyTaxRecords) — MEMBER 角色查看自己

    // 申报管理 — OWNER/ADMIN
    authGroup.GET("/tax/declarations", h.ListDeclarations)
    authGroup.GET("/tax/declarations/current", h.GetCurrentDeclaration) — 获取/创建当月申报

    // 导出 — OWNER/ADMIN
    authGroup.GET("/tax/declarations/export-excel", middleware.RequireRole("owner", "admin"), h.ExportDeclarationExcel)
    ```

    func generateDeclarationExcel(records []TaxRecord, year, month int) ([]byte, error):
    Excel 格式对齐自然人电子税务局批量导入模板 (per D-20 per research section 5):

    Sheet 名: "个税申报表_{year}年{month}月"
    表头行 (加粗蓝底白字样式):
    | 纳税人姓名 | 证件类型 | 证件号码(脱敏) | 所得项目 | 收入额 | 基本减除费用 | 专项扣除(社保) | 专项附加扣除 | 其他扣除 | 应纳税所得额 | 税率 | 速算扣除数 | 应扣税额 | 已扣税额 | 本期应扣缴税额 |

    数据行:
    - 纳税人姓名: record.EmployeeName
    - 证件类型: "居民身份证" (固定值)
    - 证件号码(脱敏): 需要从员工信息获取，暂留空或显示 "***" (V1.0 简化)
    - 所得项目: "工资薪金所得" (固定值)
    - 收入额: record.GrossIncome
    - 基本减除费用: record.BasicDeduction
    - 专项扣除(社保): record.SIDeduction
    - 专项附加扣除: record.SpecialDeduction
    - 其他扣除: 0
    - 应纳税所得额: record.CumulativeTaxableIncome (注意: 累计值)
    - 税率: record.TaxRate (转为百分比显示, 如 3%)
    - 速算扣除数: record.QuickDeduction
    - 应扣税额: record.CumulativeTax
    - 已扣税额: record.CumulativeTax - record.MonthlyTax
    - 本期应扣缴税额: record.MonthlyTax

    最后一行: 合计行（SUM公式）
    自适应列宽: 所有列宽16
    数值列格式: 保留2位小数
