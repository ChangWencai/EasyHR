```go
package socialinsurance

import (
    "encoding/json"
    "github.com/hibiken/asynq"
)

const (
    TypeMonthlyPaymentGen     = "si:monthly:gen"
    TypeStatusTransition     = "si:payment:status-transition"
)

type MonthlyPaymentGenPayload struct {
    OrgID int64 `json:"org_id"`
    Year  int   `json:"year"`
    Month int   `json:"month"`
}

type StatusTransitionPayload struct {
    OrgID int64 `json:"org_id"`
}

func NewMonthlyPaymentGenTask(payload *MonthlyPaymentGenPayload) (*asynq.Task, error) {
    data, err := json.Marshal(payload)
    if err != nil { return nil, err }
    return asynq.NewTask(TypeMonthlyPaymentGen, data), nil
}

func NewStatusTransitionTask(payload *StatusTransitionPayload) (*asynq.Task, error) {
    data, err := json.Marshal(payload)
    if err != nil { return nil, err }
    return asynq.NewTask(TypeStatusTransition, data), nil
}
```

1. **Worker 结构**：
   ```go
   type Worker struct {
       repo *Repository
       db   *gorm.DB
   }
   func NewWorker(repo *Repository, db *gorm.DB) *Worker {

   }
   ```

2. **StartWorker 注册函数**：
   ```go
   func StartWorker(srv *asynq.Server, redisOpt asynq.RedisClientOpt, repo *Repository, db *gorm.DB) {
       w := NewWorker(repo, db)
       mux := asynq.NewServeMux()
       mux.HandleFunc(TypeMonthlyPaymentGen, w.HandleMonthlyPaymentGen)
       mux.HandleFunc(TypeStatusTransition, w.HandleStatusTransition)
       srv.Start(mux)
   }
   ```

3. **HandleMonthlyPaymentGen**（D-SI-02）：
   - 接收 orgID + year + month
   - 查询该组织所有 active 员工（JOIN social_insurance_records WHERE status='active'）
   - 为每个员工创建 SIMonthlyPayment 记录（status=pending，金额从 SocialInsuranceRecord 计算）
   - 幂等性：先检查 `WHERE employee_id=? AND year_month=?` 是否存在，存在则跳过
   - 清理：删除超过 24 个月的 SIMonthlyPayment 记录（WHERE year_month < 当前月-24）

4. **HandleStatusTransition**（D-SI-03）：
   - 接收 orgID
   - CST 时区获取当天日期，判断 day >= 26 或 day < 26
   - day >= 26：所有 `status=pending AND paid_at IS NULL` 的记录 → status=overdue
   - day < 26：不做自动流转（由用户手动确认或代理 webhook 触发）
   - **注意：** 禁止 UPDATE 已 confirmed/normal 状态的历史记录

5. **HandleAgentPaymentWebhook**（D-SI-04）：
   - webhook 端点 `POST /si/webhook/agent-payment`
   - 接收参数：employee_id + year_month + success + transaction_id
   - 成功：更新 SIMonthlyPayment.status=normal, paid_at=NOW()
   - 失败：保持 status=pending，记录失败日志
