**导出逻辑（per UI-SPEC Component 6）：**
```typescript
const handleExport = async () => {
  exporting.value = true
  try {
    const params = exportOption.value === 'current'
      ? { page, pageSize }
      : { page: 1, pageSize: 99999 } // 含明细导出全部

    const response = await axios.get('/api/v1/socialinsurance/export', {
      params,
      responseType: 'blob'
    })

    // 复用 SalaryList.vue 导出模式
    const blob = new Blob([response.data], {
      type: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
    })
    const url = URL.createObjectURL(blob)
    const link = document.createElement('a')
    link.href = url
    link.download = `参保记录_${dayjs().format('YYYYMMDD_HHmmss')}.xlsx`
    link.click()
    URL.revokeObjectURL(url)

    showExportDialog.value = false
    ElMessage.success('导出成功')
  } catch (err) {
    ElMessage.error('导出失败')
  } finally {
    exporting.value = false
  }
}
```

**注意：** 如果后端采用服务端 Excel 生成（excelize），使用 blob 下载；如果前端采用 SheetJS 客户端导出（frontend/src/views/tool/SalaryList.vue），参考该模式。
