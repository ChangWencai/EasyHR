**文件1: socialinsurance.ts 扩展**（新增 API）

在 `frontend/src/api/socialinsurance.ts` 中追加：

```typescript
// SI Dashboard
export interface SIStatItem {
  label: string
  value: string
  trend_percent: string | null
  trend_direction: 'up' | 'down' | 'neutral'
}

export interface SIDashboardResponse {
  stats: SIStatItem[]
}

export const siApi = {
  // 新增 dashboard API
  dashboard: (params: { year: number; month: number }) =>
    request.get<SIDashboardResponse>('/social-insurance/dashboard', { params }),

  // 增员（SI-05~SI-08）
  enroll: (data: {
    employee_id: number
    start_month: string
    city_id: number
    base_amount: number
    housing_base?: number
    housing_rate?: number
  }) => request.post<void>('/social-insurance/enroll', data),

  // 停缴（SI-09~SI-13）
  stop: (data: {
    employee_id: number
    end_month: string
    reason: string
    transfer_date?: string
    housing_seal_date?: string
  }) => request.post<void>('/social-insurance/stop', data),

  // 五险分项详情（SI-20）
  getPaymentDetail: (params: { employee_id: number; year_month: string }) =>
    request.get<FiveInsDetailResponse>('/social-insurance/payment-detail', { params }),

  // 导出（SI-21）
  export: (params?: { employee_name?: string; status?: string }) =>
    request.getBlob('/social-insurance/export', { params, responseType: 'blob' }),

  // 欠缴汇总（SI-19）
  overdueSummary: () =>
    request.get<{ items: OverdueItem[]; max_item: OverdueItem | null }>('/social-insurance/overdue-summary'),

  // 缴费渠道更新（SI-14）
  updatePaymentChannel: (data: { payment_channel: string }) =>
    request.put('/social-insurance/payment-channel', data),
}

// 新增类型
export interface FiveInsDetailResponse {
  employee_name: string
  city_name: string
  base_amount: string
  year_month: string
  items: FiveInsItem[]
  other_payments: OtherPayment[]
  total_company: string
  total_personal: string
}

export interface FiveInsItem {
  name: string
  company_amount: string
  personal_amount: string
}

export interface OtherPayment {
  name: string
  amount: string
}

export interface OverdueItem {
  employee_name: string
  city_name: string
  year_month: string
  total_amount: string
}
```
