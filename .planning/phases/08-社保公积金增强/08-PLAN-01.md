const (
    PaymentStatusNormal       PaymentStatus = "normal"       // 正常
    PaymentStatusPending      PaymentStatus = "pending"      // 待缴
    PaymentStatusOverdue      PaymentStatus = "overdue"      // 欠缴

)

type SIMonthlyPayment struct {
    BaseModel
    EmployeeID     uint            `gorm:"not null;index:idx_org_employee_month,priority:2"`
    YearMonth      string          `gorm:"not null;size:6;index:idx_org_employee_month,priority:3"`
    Status         PaymentStatus   `gorm:"not null;default:pending"`
    PaymentChannel string          `gorm:"not null;default:self"` // self/agent_new/agent_existing
    CompanyAmount  decimal.Decimal `gorm:"type:decimal(12,2);not null;default:0"`
    PersonalAmount decimal.Decimal `gorm:"type:decimal(12,2);not null;default:0"`
    TotalAmount    decimal.Decimal `gorm:"type:decimal(12,2);not null;default:0"`
    OrgID          uint            `gorm:"not null;index:idx_org_employee_month,priority:1"`

    // 索引：唯一约束 (org_id + employee_id + year_month)，WHERE deleted_at IS NULL
    Employee       *Employee       `gorm:"foreignKey:EmployeeID"`
}

// 复合唯一索引（用 GORM index tag 实现，partial index 效果通过软删除保证）
// GORM AutoMigrate 会创建 index:idx_org_employee_month,priority:1,2,3,unique 的索引

SIDashboard 返回值结构（per D-SI-05 + D-SI-06）：

```go
type SIDashboard struct {
    TotalDue       DashboardStat     // 应缴总额（元）
    CompanyTotal   DashboardStat     // 单位部分合计（元）
    PersonalTotal  DashboardStat     // 个人部分合计（元）
    OverdueAmount DashboardStat     // 欠缴金额（元）— 带 FF5630 标记
    OverdueItems  []OverdueItem     `json:"overdueItems"` // 欠缴列表（用于红色横幅 per D-SI-10）
}

// OverdueItem：欠缴横幅数据结构
type OverdueItem struct {
    ID          uint   `json:"id"`
    EmployeeID  uint   `json:"employeeID"`
    EmployeeName string `json:"employeeName"` // 从 Employee.Name 关联获取
    City        string `json:"city"`          // 缴费城市
    YearMonth   string `json:"yearMonth"`     // 欠缴月（YYYYMM）
    Amount      string `json:"amount"`        // 格式 "¥X,XXX.XX"
}

// DashboardStat 结构：
type DashboardStat struct {
    Value       string `json:"value"`       // 格式 "¥X,XXX,XXX.XX"
    RawValue    decimal.Decimal `json:"rawValue"`
    Trend       *TrendData      `json:"trend"`        // nil if 上月无数据
}

// TrendData：
type TrendData struct {
    Percent  float64 `json:"percent"`  // 环比百分比，如 12.5
    Direction string `json:"direction"` // "up"/"down"/"neutral"
    Label    string  `json:"label"`     // "+12.5%" 或 "-12.5%"
}
```

环比计算逻辑（per D-SI-06）：
- 遍历当月和上月 SIMonthlyPayment 记录（status=normal/pending/overdue）
- 仅统计 confirmed=true 的月份
- 上月无数据时，Trend = nil（前端显示"—"）
- 欠缴金额Trend：overdue增加=bad，显示红色

参考 internal/salary/dashboard_service.go 的 errgroup 并发模式。
