### Step 3: utils/request.js

```javascript
const API_BASE = 'https://api.easyhr.example.com/api/v1'

function request({ url, method = 'GET', data, needAuth = true }) {
  return new Promise((resolve, reject) => {
    const header = { 'Content-Type': 'application/json' }
    if (needAuth) {
      const token = wx.getStorageSync('token')
      if (!token) {
        wx.redirectTo({ url: '/pages/login/login' })
        return reject(new Error('no token'))
      }
      header['Authorization'] = `Bearer ${token}`
    }
    wx.request({
      url: API_BASE + url,
      method,
      data,
      header,
      success: (res) => {
        if (res.statusCode === 401) {
          wx.removeStorageSync('token')
          wx.redirectTo({ url: '/pages/login/login' })
          return reject(new Error('unauthorized'))
        }
        if (res.data.code !== 0) {
          wx.showToast({ title: res.data.message || '请求失败', icon: 'none' })
          return reject(new Error(res.data.message))
        }
        resolve(res.data.data)
      },
      fail: () => {
        wx.showToast({ title: '网络错误', icon: 'none' })
        reject(new Error('network error'))
      }
    })
  })
}

module.exports = { request }
```

```go
// ExportToExcel exports trial balance / ledger / account balance to .xlsx
func (s *BookService) ExportToExcel(ctx context.Context, orgID int64, req BookExportRequest) ([]byte, error) {
    f := excelize.NewFile()
    // Sheet 1: Trial Balance
    ws, _ := f.NewSheet("科目余额表")
    // Headers
    headers := []string{"科目编码", "科目名称", "期初余额", "借方发生", "贷方发生", "期末余额"}
    for i, h := range headers {
        cell, _ := excelize.CoordinatesToCellName(i+1, 1)
        f.SetCellValue("科目余额表", cell, h)
    }
    // Data: call GetTrialBalance, iterate rows
    tb, err := s.GetTrialBalance(ctx, orgID, req.PeriodID)
    if err != nil {
        return nil, err
    }
    row := 2
    for _, item := range tb.Items {
        f.SetCellValue("科目余额表", fmt.Sprintf("A%d", row), item.Code)
        f.SetCellValue("科目余额表", fmt.Sprintf("B%d", row), item.Name)
        f.SetCellValue("科目余额表", fmt.Sprintf("C%d", row), item.OpenBalance.String())
        f.SetCellValue("科目余额表", fmt.Sprintf("D%d", row), item.Debit.String())
        f.SetCellValue("科目余额表", fmt.Sprintf("E%d", row), item.Credit.String())
        f.SetCellValue("科目余额表", fmt.Sprintf("F%d", row), item.CloseBalance.String())
        row++
    }
    f.DeleteSheet("Sheet1")
    buf, err := f.WriteToBuffer()
    if err != nil {
        return nil, err
    }
    return buf.Bytes(), nil
}
```

### FINC-12: Book Export Route (handler_book.go)

```go
func (h *BookHandler) ExportBook(c *gin.Context) {
    var req struct {
        PeriodID int64 `form:"period_id" binding:"required"`
        Type     string `form:"type"` // trial_balance, ledger, account_balance
    }
    if err := c.ShouldBindQuery(&req); err != nil {
        c.Error(err)
        return
    }
    orgID := c.GetInt64("org_id")
    data, err := h.bookSvc.ExportToExcel(c.Request.Context(), orgID, req.PeriodID, req.Type)
    if err != nil {
        c.Error(err)
        return
    }
    c.Header("Content-Disposition", "attachment; filename=账簿导出.xlsx")
    c.Data(200, "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet", data)
}
```
