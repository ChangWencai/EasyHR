// ScanUrgencyStatus 扫描限时任务更新 urgency_status
// 执行：UPDATE todo_items SET urgency_status = CASE WHEN ... WHERE is_time_limited=true AND status NOT IN ('completed','terminated')
func (r *Repository) ScanUrgencyStatus(ctx context.Context) (int, error) {
    // D-09-04 规则：
    // deadline - now > 7 days: normal
    // deadline - now in [-15, 7]: overdue
    // deadline - now < -15: expired
    // 仅更新限时、未完成的记录

    now := time.Now()

    // 更新 expired: 超期超过15天
    result1 := r.db.Model(&TodoItem{}).
        Where("is_time_limited = ? AND status NOT IN ? AND deadline IS NOT NULL AND deadline < ?",
            true, []string{TodoStatusCompleted, TodoStatusTerminated}, now.AddDate(0, 0, -15)).
        Update("urgency_status", UrgencyExpired)
    if result1.Error != nil {
        return 0, result1.Error
    }

    // 更新 overdue: 超期1-15天 或 剩余<=7天
    result2 := r.db.Model(&TodoItem{}).
        Where("is_time_limited = ? AND status NOT IN ? AND deadline IS NOT NULL AND urgency_status != ?",
            true, []string{TodoStatusCompleted, TodoStatusTerminated}, UrgencyOverdue).
        Where("deadline <= ? OR deadline >= ?",
            now.AddDate(0, 0, 7), now.AddDate(0, 0, -15)).
        Update("urgency_status", UrgencyOverdue)
    if result2.Error != nil {
        return 0, result2.Error
    }

    // 更新 normal: 剩余>7天
    result3 := r.db.Model(&TodoItem{}).
        Where("is_time_limited = ? AND status NOT IN ? AND urgency_status != ?",
            true, []string{TodoStatusCompleted, TodoStatusTerminated}, UrgencyNormal).
        Where("deadline > ?", now.AddDate(0, 0, 7)).
        Update("urgency_status", UrgencyNormal)
    if result3.Error != nil {
        return 0, result3.Error
    }

    return int(result1.RowsAffected + result2.RowsAffected + result3.RowsAffected), nil
}

// UpdateCarouselActivation 根据 start_at/end_at 更新轮播图 active 状态
func (r *Repository) UpdateCarouselActivation(ctx context.Context) (int, error) {
    now := time.Now()
    // 激活：时间区间内且 active=true
    activate := r.db.Model(&CarouselItem{}).
        Where("active = ? AND (start_at IS NULL OR start_at <= ?) AND (end_at IS NULL OR end_at >= ?)",
            false, now, now).
        Update("active", true)
    if activate.Error != nil {
        return 0, activate.Error
    }

    // 停用：超出时间区间
    deactivate := r.db.Model(&CarouselItem{}).
        Where("(start_at IS NOT NULL AND start_at > ?) OR (end_at IS NOT NULL AND end_at < ?)",
            now, now).
        Update("active", false)
    if deactivate.Error != nil {
        return 0, deactivate.Error
    }

    return int(activate.RowsAffected + deactivate.RowsAffected), nil
}

// GenerateMonthlyTodos 生成每月例行待办（个税申报+社保缴费，，每月1日触发）

    cst := time.FixedZone("CST", 8*3600)
    today := time.Now().In(cst)
    deadline := time.Date(today.Year(), today.Month(), 15, 23, 59, 59, 0, cst)

    // 获取所有活跃组织
    var orgIDs []int64
    r.db.Model(&struct{ OrgID int64 }{}).Table("organizations").Pluck("id", &orgIDs)

    for _, orgID := range orgIDs {
        // 个税申报
        taxDeadline := deadline
        taxTodo := &TodoItem{
            OrgID:         orgID,
            Title:         fmt.Sprintf("%d月个税申报，请于15日前完成", today.Month()),
            Type:          TodoTypeTaxDeclaration,
            Deadline:      &taxDeadline,
            IsTimeLimited: true,
            Status:        TodoStatusPending,
            UrgencyStatus: UrgencyNormal,
            SourceType:    "tax",

        _ = r.CreateTodo(ctx, taxTodo) // 幂等由 CreateTodo 处理

// InviteTodo 创建协办邀请（生成Token）
type InviteResult struct {
    URL string `json:"url"`
}
func (s *Service) InviteTodo(ctx context.Context, orgID int64, todoID int64, userID int64) (*InviteResult, error) {
    // 验证待办存在
    todo, err := s.repo.FindTodoByID(ctx, orgID, todoID)
    if err != nil {
        return nil, fmt.Errorf("todo not found")
    }

    token, err := GenerateInviteToken()
    if err != nil {
        return nil, fmt.Errorf("generate token: %w", err)
    }

    now := time.Now()
    invite := &TodoInvite{
        OrgID:     orgID,
        TodoID:   todoID,
        Token:    token,
        Status:   InviteStatusPending,
        CreatedBy: userID,
        ExpiresAt: now.Add(InviteExpiryDuration),
    }
    if err := s.repo.CreateInvite(ctx, invite); err != nil {
        return nil, fmt.Errorf("create invite: %w", err)
    }

    // 返回邀请URL
    inviteURL := fmt.Sprintf("/todo/%d/invite?token=%s", todoID, token)
    return &InviteResult{URL: inviteURL}, nil
}

// TerminateTodo 终止待办任务（状态变为 terminated）
func (s *Service) TerminateTodo(ctx context.Context, orgID int64, todoID int64) error {
    return s.repo.UpdateTodoStatus(ctx, orgID, todoID, TodoStatusTerminated)
}

// VerifyInviteToken 验证协办邀请Token
type VerifyResult struct {
    Valid    bool   `json:"valid"`
    Expired  bool   `json:"expired"`
    Title    string `json:"title"`
    TodoID   int64  `json:"todo_id"`
}
func (s *Service) VerifyInviteToken(ctx context.Context, token string) (*VerifyResult, error) {
    invite, err := s.repo.FindInviteByToken(ctx, token)
    if err != nil {
        return &VerifyResult{Valid: false}, nil // 找不到Token → 无效
    }

    if invite.Status == InviteStatusUsed {
        return &VerifyResult{Valid: false}, nil
    }
    if time.Now().After(invite.ExpiresAt) {
        return &VerifyResult{Valid: false, Expired: true}, nil
    }

    // 查询关联待办标题
    todo, err := s.repo.FindTodoByID(ctx, invite.OrgID, invite.TodoID)
    if err != nil {
        return &VerifyResult{Valid: false}, nil
    }

    return &VerifyResult{
        Valid:  true,
        Title:  todo.Title,
        TodoID: invite.TodoID,
    }, nil
}

// InviteTodo handles POST /api/v1/todos/:id/invite
func (h *Handler) InviteTodo(c *gin.Context) {
    orgIDVal, exists := c.Get("org_id")
    if !exists {
        response.Error(c, http.StatusUnauthorized, 40100, "missing org_id")
        return
    }
    userIDVal, _ := c.Get("user_id")
    orgID := orgIDVal.(int64)
    userID, _ := userIDVal.(int64)

    idStr := c.Param("id")
    todoID, err := strconv.ParseInt(idStr, 10, 64)
    if err != nil {
        response.Error(c, http.StatusBadRequest, 40000, "invalid id")
        return
    }

    result, err := h.svc.InviteTodo(c.Request.Context(), orgID, todoID, userID)
    if err != nil {
        response.Error(c, http.StatusInternalServerError, 50000, "invite failed: "+err.Error())
        return
    }

    response.Success(c, result)
}

// TerminateTodo handles PUT /api/v1/todos/:id/terminate

    if !exists {
        response.Error(c, http.StatusUnauthorized, 40100, "missing org_id")

    }
    orgID := orgIDVal.(int64)

    idStr := c.Param("id")
    todoID, err := strconv.ParseInt(idStr, 10, 64)
    if err != nil {
        response.Error(c, http.StatusBadRequest, 40000, "invalid id")

        response.Error(c, http.StatusInternalServerError, 50000, "terminate failed: "+err.Error())

    }

    response.Success(c, gin.H{"id": todoID})
}

// VerifyInviteToken handles GET /api/v1/todos/invite/:token
// Public endpoint — no auth required
func (h *Handler) VerifyInviteToken(c *gin.Context) {
    token := c.Param("token")
    if token == "" {
        response.Error(c, http.StatusBadRequest, 40000, "missing token")
        return
    }

    result, err := h.svc.VerifyInviteToken(c.Request.Context(), token)
    if err != nil {
        response.Error(c, http.StatusInternalServerError, 50000, "verify failed: "+err.Error())
        return
    }

    if !result.Valid {
        if result.Expired {
            response.Error(c, http.StatusGone, 41001, "邀请链接已过期")
            return
        }
        response.Error(c, http.StatusNotFound, 40400, "邀请链接无效")
        return
    }

    response.Success(c, result)
}

// GetInviteTodo handles GET /api/v1/todos/invite/:token/detail — public
func (h *Handler) GetInviteTodo(c *gin.Context) {
    token := c.Param("token")
    result, err := h.svc.VerifyInviteToken(c.Request.Context(), token)
    if err != nil || !result.Valid {
        response.Error(c, http.StatusNotFound, 40400, "邀请不存在")
        return
    }
    response.Success(c, result)
}

package todo

import (
	"context"
	"fmt"
	"log"
	"time"

	"github.com/go-co-op/gocron/v2"
	"github.com/redis/go-redis/v9"

// Scheduler 定时任务调度器
type Scheduler struct {
	repo *Repository
	rdb  *redis.Client
}

// NewScheduler 创建调度器
func NewScheduler(repo *Repository, rdb *redis.Client) *Scheduler {
	return &Scheduler{repo: repo, rdb: rdb}
}

// Start 启动调度器
func (s *Scheduler) Start() (gocron.Scheduler, error) {
	opts := []gocron.SchedulerOption{
		gocron.WithLocation(time.FixedZone("CST", 8*3600)),
	}

	// Redis 可用时启用分布式锁（复用 socialinsurance 的 redisLocker）
	if s.rdb != nil {
		locker := newTodoRedisLocker(s.rdb)
		opts = append(opts, gocron.WithDistributedLocker(locker))
	}

	sched, err := gocron.NewScheduler(opts...)
	if err != nil {
		return nil, fmt.Errorf("create scheduler: %w", err)
	}

	// 每日 02:00 CST — 扫描 urgency_status
	_, err = sched.NewJob(
		gocron.DailyJob(1, gocron.NewAtTimes(gocron.NewAtTime(2, 0, 0))),
		gocron.NewTask(func() {

type todoRedisLocker struct {
	rdb *redis.Client
}

func newTodoRedisLocker(rdb *redis.Client) *todoRedisLocker {
	return &todoRedisLocker{rdb: rdb}
}

func (l *todoRedisLocker) Lock(ctx context.Context, key string) (gocron.Lock, error) {
	lockKey := "easyhr:todo:lock:" + key
	ok, err := l.rdb.SetNX(ctx, lockKey, "locked", 60*time.Second).Result()
	if err != nil {
		return nil, err
	}
	if !ok {
		return nil, fmt.Errorf("job already locked: %s", key)
	}
	return &todoRedisLock{rdb: l.rdb, key: lockKey}, nil
}

type todoRedisLock struct {
	rdb *redis.Client
	key string
}

func (l *todoRedisLock) Unlock(ctx context.Context) error {
	return l.rdb.Del(ctx, l.key).Err()
}

func (s *ContractService) CheckContractRenewalReminders(ctx context.Context) error {
    // 查找合同到期日在未来30天内的 active 合同
    now := time.Now()
    deadline := now.AddDate(0, 1, 0) // 30天后

    contracts, err := s.repo.FindContractsExpiringSoon(ctx, now, deadline)
    if err != nil {
        return fmt.Errorf("find expiring contracts: %w", err)
    }

    for _, contract := range contracts {
        // 幂等检查：是否已存在该合同续签待办
        exists, _ := s.todoRepo.ExistsBySource(ctx, contract.OrgID, todo.TodoTypeContractRenew, contract.ID)
        if exists {
            continue
        }

        // 获取员工姓名
        emp, _ := s.repo.FindEmployeeByID(ctx, contract.OrgID, contract.EmployeeID)
        empName := ""
        if emp != nil {
            empName = emp.Name
        }

        renewalTodo := &todo.TodoItem{
            OrgID:         contract.OrgID,
            Title:         fmt.Sprintf("员工 %s 的劳动合同即将到期（%s），请及时处理续签", empName, contract.EndDate.Format("2006-01-02")),
            Type:          todo.TodoTypeContractRenew,
            EmployeeID:    &contract.EmployeeID,
            EmployeeName:  empName,
            Deadline:      contract.EndDate,
            IsTimeLimited: true,
            Status:        todo.TodoStatusPending,
            UrgencyStatus: todo.UrgencyNormal,
            SourceType:    "contract",
            SourceID:      &contract.ID,
            CreatorName:   "系统",
        }
        _ = s.todoRepo.CreateTodo(ctx, renewalTodo)
    }

    return nil
}

// FindContractsExpiringSoon 查找即将到期的合同
func (s *ContractService) FindContractsExpiringSoon(ctx context.Context, now, deadline time.Time) ([]Contract, error) {
    var contracts []Contract
    err := s.db.Model(&Contract{}).
        Where("status = ? AND end_date IS NOT NULL AND end_date >= ? AND end_date <= ?",
            ContractStatusActive, now, deadline).
        Find(&contracts).Error
    return contracts, err
}

// FindEmployeeByID 查找员工
func (s *ContractService) FindEmployeeByID(ctx context.Context, orgID, empID int64) (*Employee, error) {
    var emp Employee
    err := s.db.Where("org_id = ? AND id = ?", orgID, empID).First(&emp).Error
    if err != nil {
        return nil, err
    }
    return &emp, nil
}
