    5. **创建 internal/employee/pdf.go** — 合同 PDF 模板生成（per D-18, D-21）:
       ```go
       package employee

       import (
           "bytes"
           "fmt"
           "time"
           "github.com/go-pdf/fpdf"
       )

       // ContractPDFData 合同 PDF 填充数据
       type ContractPDFData struct {
           OrgName       string
           CreditCode    string
           EmployeeName  string
           IDCard        string
           Position      string
           City          string
           Salary        float64
           StartDate     time.Time
           EndDate       *time.Time
           ContractType  string
           SignDate      time.Time
       }

       // GenerateContractPDF 生成劳动合同 PDF
       func GenerateContractPDF(data *ContractPDFData) ([]byte, error) {
           pdf := fpdf.New("P", "mm", "A4", "")
           pdf.AddPage()
           pdf.SetFont("Helvetica", "B", 20) // V1.0 使用 Helvetica 英文字体
           // 注意: 中文字体需要注册 TTF 文件，V1.0 先用 Helvetica + 英文标注关键信息

           // 标题
           pdf.CellFormat(0, 15, "Labor Contract", "", 0, "C", false, 0, "")
           pdf.Ln(20)

           pdf.SetFont("Helvetica", "", 12)
           lineHeight := 8.0

           // 甲方信息
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Party A (Employer): %s", data.OrgName), "", 1, "L", false, 0, "")
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Credit Code: %s", data.CreditCode), "", 1, "L", false, 0, "")

           pdf.Ln(5)

           // 乙方信息
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Party B (Employee): %s", data.EmployeeName), "", 1, "L", false, 0, "")
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("ID Card: %s", data.IDCard), "", 1, "L", false, 0, "")

           pdf.Ln(5)

           // 合同条款
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Position: %s", data.Position), "", 1, "L", false, 0, "")
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Work Location: %s", data.City), "", 1, "L", false, 0, "")
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Monthly Salary: %.2f CNY", data.Salary), "", 1, "L", false, 0, "")

           contractPeriod := fmt.Sprintf("From %s", data.StartDate.Format("2006-01-02"))
           if data.EndDate != nil {
               contractPeriod += fmt.Sprintf(" to %s", data.EndDate.Format("2006-01-02"))
           } else {
               contractPeriod += " (Indefinite term)"
           }
           pdf.CellFormat(0, lineHeight, fmt.Sprintf("Contract Period: %s", contractPeriod), "", 1, "L", false, 0, "")

           pdf.Ln(20)

           // 签名区域
           pdf.CellFormat(0, lineHeight, "Party A Seal: ____________    Date: ____________", "", 1, "L", false, 0, "")
           pdf.Ln(5)
           pdf.CellFormat(0, lineHeight, "Party B Signature: ____________    Date: ____________", "", 1, "L", false, 0, "")

           // 输出到 buffer
           var buf bytes.Buffer
           if err := pdf.Output(&buf); err != nil {
               return nil, fmt.Errorf("generate PDF: %w", err)
           }
           return buf.Bytes(), nil
       }
       ```

       **关于中文字体的处理策略:**
       - V1.0 fpdf 使用内置 Helvetica 字体，中文内容以拼音/英文标注
       - executor 应在 pdf.go 中添加一个 TODO 注释说明后续需注册中文字体
       - 测试时验证 PDF 以 `%PDF-` 开头即可
       - 如要支持中文，可后续嵌入思源黑体 TTF 文件（~16MB），通过 `pdf.AddUTF8Font()` 注册

       关键方法:
       - `CreateContract(ctx, orgID, userID, employeeID int64, req *CreateContractRequest) (*ContractResponse, error)`:
         1. empRepo.FindByID 验证员工存在
         2. 构建 Contract{Status: "draft"}
         3. contractRepo.Create
         4. 返回 ContractResponse

       - `GeneratePDF(ctx, orgID, contractID int64) ([]byte, error)`:
         1. contractRepo.FindByID(orgID, contractID)
         2. empRepo.FindByID(orgID, contract.EmployeeID) 获取员工信息（含解密身份证号）
         3. 查询 Organization 信息（直接用 db.First）
         4. 构建 ContractPDFData，填充企业和员工信息
         5. 调用 GenerateContractPDF
         6. 更新合同 status → "pending_sign"
         7. 返回 PDF []byte

       - `UploadSigned(ctx, orgID, contractID int64, req *UploadSignedRequest) (*ContractResponse, error)`:
         1. contractRepo.FindByID(orgID, contractID)
         2. 更新 signed_pdf_url, sign_date, status → "signed"
         3. 如果 sign_date <= end_date（或无固定期限），status → "active"
         4. 返回 ContractResponse

       - `TerminateContract(ctx, orgID, contractID int64, req *TerminateContractRequest) (*ContractResponse, error)`:
         1. contractRepo.FindByID(orgID, contractID)
         2. 更新 terminate_date, terminate_reason, status → "terminated"
         3. 返回 ContractResponse

       - `GetContract(ctx, orgID, contractID int64) (*ContractResponse, error)`: 查询详情含员工姓名
       - `ListByEmployee(ctx, orgID, employeeID int64, page, pageSize int) ([]ContractResponse, int64, error)`: 按员工查合同
       - `ListContracts(ctx, orgID int64, status string, page, pageSize int) ([]ContractResponse, int64, error)`: 按企业查所有合同
       - `toContractResponse(c *Contract, empName string) *ContractResponse`: 辅助方法

       ```go
       func (h *ContractHandler) RegisterRoutes(rg *gin.RouterGroup, authMiddleware gin.HandlerFunc) {
           authGroup := rg.Group("")
           authGroup.Use(authMiddleware)

           // 合同 CRUD — OWNER/ADMIN
           authGroup.POST("/employees/:id/contracts", middleware.RequireRole("owner", "admin"), h.CreateContract)
           authGroup.GET("/employees/:id/contracts", h.ListByEmployee) // 所有角色可查看
           authGroup.GET("/contracts/:id", h.GetContract)              // 所有角色可查看
           authGroup.PUT("/contracts/:id", middleware.RequireRole("owner", "admin"), h.UpdateContract)
           authGroup.POST("/contracts/:id/generate-pdf", middleware.RequireRole("owner", "admin"), h.GeneratePDF)
           authGroup.POST("/contracts/:id/upload-signed", middleware.RequireRole("owner", "admin"), h.UploadSigned)
           authGroup.PUT("/contracts/:id/terminate", middleware.RequireRole("owner", "admin"), h.TerminateContract)
       }
       ```
