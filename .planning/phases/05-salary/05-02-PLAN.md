    ```go
    package salary

    import (
        "math"
        "github.com/wencai/easyhr/internal/tax"
    )

    // PayrollItemInput 核算输入项
    type PayrollItemInput struct {
        ItemName string
        ItemType string  // income/deduction
        Amount   float64
    }

    // PayrollResult 核算结果
    type PayrollResult struct {
        GrossIncome     float64
        SIDeduction     float64
        Tax             float64
        TotalDeductions float64  // 扣款项合计（事假/病假/其他扣款）
        NetIncome       float64
        TaxResult       *tax.TaxResult // 完整个税结果（用于审计）
    }

    // AbnormalCheck 异常发放检查结果
    type AbnormalCheck struct {
        EmployeeID    int64
        EmployeeName  string
        CurrentNet    float64
        PreviousNet   float64
        DeviationRate float64 // 偏差百分比
    }

    // calculateDailyWage 日薪计算（per D-14: 月基本工资 / 21.75）
    func calculateDailyWage(monthlyBase float64) float64 {
        return roundTo2(monthlyBase / 21.75)
    }

    // calculateLeaveDeduction 请假扣款计算
    func calculateLeaveDeduction(dailyWage float64, days float64) float64 {
        return roundTo2(dailyWage * days)
    }

    // calculatePayroll 工资核算纯函数（per D-06）
    // 输入：薪资项列表、社保个人扣款、个税结果
    // 输出：完整核算结果
    func calculatePayroll(items []PayrollItemInput, siDeduction float64, taxResult *tax.TaxResult) *PayrollResult {
        var grossIncome float64
        var totalDeductions float64

        for _, item := range items {
            if item.ItemType == "income" {
                grossIncome = roundTo2(grossIncome + item.Amount)
            } else {
                totalDeductions = roundTo2(totalDeductions + item.Amount)
            }
        }

        monthlyTax := float64(0)
        if taxResult != nil {
            monthlyTax = taxResult.MonthlyTax
        }

        netIncome := roundTo2(grossIncome - siDeduction - monthlyTax - totalDeductions)

        return &PayrollResult{
            GrossIncome:     grossIncome,
            SIDeduction:     siDeduction,
            Tax:             monthlyTax,
            TotalDeductions: totalDeductions,
            NetIncome:       netIncome,
            TaxResult:       taxResult,
        }
    }

    // checkAbnormalPayments 异常发放检查（per D-17: 偏差>30% 标记为异常）
    func checkAbnormalPayments(currentRecords []struct{ EmployeeID int64; EmployeeName string; NetIncome float64 }, previousRecords map[int64]float64) []AbnormalCheck {
        var abnormal []AbnormalCheck
        for _, r := range currentRecords {
            prevNet, exists := previousRecords[r.EmployeeID]
            if !exists || prevNet == 0 {
                continue // 无上月数据或上月为0，不检查
            }
            deviation := math.Abs(r.NetIncome-prevNet) / prevNet
            if deviation > 0.30 {
                abnormal = append(abnormal, AbnormalCheck{
                    EmployeeID:    r.EmployeeID,
                    EmployeeName:  r.EmployeeName,
                    CurrentNet:    r.NetIncome,
                    PreviousNet:   prevNet,
                    DeviationRate: roundTo2(deviation * 100),
                })
            }
        }
        return abnormal
    }

    // roundTo2 保留两位小数
    func roundTo2(val float64) float64 {
        return math.Round(val*100) / 100
    }
    ```

    **1. 扩展 internal/salary/service.go** — 新增核心核算方法：

    - `CreatePayroll(orgID, userID int64, year, month int, copyFromMonth *string) ([]PayrollRecordResponse, error)`
      - 创建指定月份所有在职员工的 PayrollRecord（status=draft）
      - 如果 copyFromMonth 非空，复制上月 PayrollItem 数据到当月（per D-07）
      - 为每个员工创建空的 PayrollItem（基于启用的薪资项模板）
      - 新入职员工无上月数据时，基本工资从 Contract.Salary 同步（通过 EmployeeProvider.GetEmployeeByID）
      - 离职员工不纳入（EmployeeProvider.GetActiveEmployees 只返回在职员工）

    - `CalculatePayroll(orgID, userID int64, year, month int) (*BatchCalculateResponse, error)`
      - 查询当月所有 draft/calculated 状态的 PayrollRecord
      - 对每个员工：
        1. 获取 SalaryItem 列表（当月各项金额）
        2. 调用 TaxProvider.CalculateTax(orgID, empID, year, month, grossIncome) 获取个税
        3. 调用 SIDeductionProvider.GetPersonalDeduction(orgID, empID, monthStr) 获取社保扣款
        4. 调用 calculatePayroll 纯函数计算结果
        5. 更新 PayrollRecord 字段（GrossIncome/SIDeduction/Tax/TotalDeductions/NetIncome）
        6. 更新/创建 PayrollItem 快照（记录各项名称和金额）
        7. 状态更新为 calculated
      - 返回核算汇总

    - `GetPayrollList(orgID int64, year, month, page, pageSize int) ([]PayrollRecordResponse, int64, error)`
      - 查询指定月份所有 PayrollRecord 列表（含 PayrollItem 明细）

    - `GetPayrollDetail(orgID, recordID int64) (*PayrollRecordDetailResponse, error)`
      - 查询单个 PayrollRecord + PayrollItem 明细

    - `ConfirmPayroll(orgID, userID int64, year, month int) (*ConfirmResponse, error)`
      - 校验所有 PayrollRecord 状态必须为 calculated
      - 检查异常发放：对比上月实发，偏差>30% 的员工列表返回（per D-17）
      - 状态更新为 confirmed
      - 确认后对每个员工调用 BaseAdjustmentProvider.SuggestBaseAdjustment（per D-08）
      - 返回 ConfirmResponse（含异常列表，不阻断操作）

    - `RecordPayment(orgID, userID, recordID int64, req *RecordPaymentRequest) error`
      - 校验 PayrollRecord 状态必须为 confirmed
      - 更新 PayMethod/PayDate/PayNote
      - 状态更新为 paid

    - `ImportAttendance(orgID, userID int64, year, month int, file []byte) (*AttendanceImportResult, error)`
      - 解析 Excel 文件（固定模板：员工姓名、事假(天)、病假(天)、备注）
      - 按姓名匹配员工（精确匹配，per D-13）
      - 计算扣款：事假扣款 = (基本工资/21.75) * 事假天数，病假扣款同理（per D-14）
      - 自动填入当月 PayrollItem 的事假扣款和病假扣款项（per D-15）
      - 未匹配的行标记为错误，返回 AttendanceImportResult
      - 文件大小限制 5MB，行数限制 200 行
