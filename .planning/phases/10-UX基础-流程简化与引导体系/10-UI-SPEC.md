## Spacing Scale

Source: `frontend/src/styles/variables.scss` CSS custom properties. All values are multiples of 4.

| Token | Value | Usage |
|-------|-------|-------|
| xs | 4px | Icon gaps, inline padding |
| sm | 8px | Compact element spacing, gap utilities |

## useMessage() Composable API

```typescript
// File: frontend/src/composables/useMessage.ts

interface MessageOptions {
  showActions?: boolean   // show retry/contact-admin buttons (error only)
  duration?: number      // override default duration
}

export function useMessage() {
  function success(msg: string, opts?: MessageOptions): void
  function error(msg: string, opts?: MessageOptions): void   // default: no auto-close, showActions=true
  function warning(msg: string, opts?: MessageOptions): void // default: duration=3000
  function info(msg: string, opts?: MessageOptions): void     // default: duration=2000
}

// Duration defaults:
//   success / info: 2000ms (auto-close)
//   warning: 3000ms (auto-close)
//   error: 0 (no auto-close, show retry + contact-admin buttons)
```

## Tour State Management

```typescript
// localStorage key
const TOUR_DONE_KEY = 'hasSeenTour'  // D-10-09

// Tour visibility logic
if (!localStorage.getItem(TOUR_DONE_KEY)) {
  showTour = true
}

// On tour complete or skip
function completeTour() {
  localStorage.setItem(TOUR_DONE_KEY, 'true')
  showTour = false
}
```
