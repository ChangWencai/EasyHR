```typescript
export interface TourStep {
  title: string
  body: string
  target?: string   // CSS selector for target element to highlight

  steps: TourStep[]
```

**State:**
```typescript
const currentStep = ref(0)
const TOUR_DONE_KEY = 'hasSeenTour'

const hasSeenTour = localStorage.getItem(TOUR_DONE_KEY)

watch(() => props.visible, (val) => {
  if (!val) currentStep.value = 0
})

function next() {
  if (currentStep.value < props.steps.length - 1) {
    currentStep.value++
  } else {
    complete()
  }
}

function prev() {
  if (currentStep.value > 0) currentStep.value--
}

function skip() {
  complete()
}

function complete() {
  localStorage.setItem(TOUR_DONE_KEY, 'true')
  emit('update:visible', false)
  emit('complete')
}
```

**Tooltip positioning (per D-10-07):**
```typescript
function getTooltipStyle(target?: string): Record<string, string> {
  if (!target) {
    return { top: '40%', left: '50%', transform: 'translate(-50%, -50%)' }
  }
  // Find target element and position tooltip near it
  const el = document.querySelector(target)
  if (!el) {
    return { top: '40%', left: '50%', transform: 'translate(-50%, -50%)' }
  }
  const rect = el.getBoundingClientRect()
  return {
    position: 'fixed',
    top: `${rect.top + rect.height / 2}px`,
    left: `${rect.right + 16}px`,
    transform: 'translateY(-50%)',
  }
}
```

```typescript
import { useMessage } from '@/composables/useMessage'

const $msg = useMessage()

// Error code → user-friendly message mapping (D-10-12)
const ERROR_MESSAGES: Record<number, string> = {
  400: '请求参数错误，请检查输入',
  401: '登录已过期，请重新登录',  // handled separately with redirect
  403: '您没有权限进行此操作',
  404: '请求的数据不存在',
  409: '数据冲突，请刷新后重试',
  422: '数据验证失败，请检查输入',
  500: '服务器异常，请稍后重试',
  502: '服务暂时不可用，请稍后重试',
  503: '服务暂时不可用，请稍后重试',
}

request.interceptors.response.use(
  (response) => response.data,
  async (error: AxiosError) => {
    const err = error as unknown as ApiError
    const status = err.response?.status
    const serverMessage = err.response?.data?.message

    // 401: redirect to login (per existing pattern)
    if (status === 401) {
      localStorage.removeItem('token')
      $msg.error('登录已过期，请重新登录')
      router.push('/login')
      return Promise.reject(error)
    }

    // Determine user message
    let userMessage = ERROR_MESSAGES[status ?? 0] ?? '操作失败，请稍后重试'

    // Network errors (no response)
    if (!status) {
      if (error.code === 'ECONNABORTED' || error.message?.includes('timeout')) {
        userMessage = '请求超时，请稍后重试'
      } else {
        userMessage = '网络连接失败，请检查网络后重试'
      }
    }

    // Show error with action buttons for retryable errors (D-10-13)
    const retryable = [500, 502, 503, 'ECONNABORTED', 'ERR_NETWORK'].includes(status ?? error.code)

    if (retryable) {
      // Show toast with retry guidance — actual retry handled by ErrorActions component in pages
      $msg.error(userMessage, { showActions: true })
    } else {
      $msg.error(userMessage)
    }

    return Promise.reject(error)
  },
)
```
