```typescript
import { ref } from 'vue'
import { departmentApi } from '@/api/department'
import { attendanceApi } from '@/api/attendance'

const selectedDepts = ref<number[]>([])
const deptOptions = ref<{ label: string; value: number }[]>([])

// Load departments on mount
onMounted(async () => {
  const depts = await departmentApi.list()
  deptOptions.value = depts.map(d => ({ label: d.name, value: d.id }))
  load(1)
})

// Format dept_ids as comma-separated string for API calls
const deptIdsParam = computed(() =>
  selectedDepts.value.length > 0 ? selectedDepts.value.join(',') : ''
)
```

```typescript
await attendanceApi.getComplianceOvertime({

  dept_ids: deptIdsParam.value,
  page: p,
  page_size: pageSize
})
```

Export function:
```typescript
async function handleExport() {
  try {
    const res = await attendanceApi.exportComplianceMonthly({
      year_month: selectedMonth.value,
      dept_ids: deptIdsParam.value
    })
    const blob = res as unknown as Blob
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `月度考勤汇总_${selectedMonth.value}.xlsx`
    a.click()
    URL.revokeObjectURL(url)
    ElMessage.success('导出成功')
  } catch { ElMessage.error('导出失败') }
}
```
