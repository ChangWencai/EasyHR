**步骤2: 在 routes 数组定义后、createRouter 调用前，添加 router.beforeEach：**
```typescript
// Auth Guard（per D-10, D-11, D-12）
router.beforeEach((to, from, next) => {
  const authStore = useAuthStore()

  // /login 本身不做守卫检查（per D-12）
  if (to.path === '/login') {
    next()
    return
  }

  // 检查是否访问 AppLayout 子路由（AppLayout 的子路由都是受保护路由）
  // AppLayout 的 path 是 '/'，其 children 下的路由都是受保护路由
  // /onboarding/org-setup 也是受保护路由（但首次登录时用户还没有 token，所以 onboarding 不做守卫）
  const isProtectedRoute =
    to.path.startsWith('/home') ||
    to.path.startsWith('/employee') ||
    to.path.startsWith('/tool') ||
    to.path.startsWith('/finance') ||
    to.path.startsWith('/mine')

  if (isProtectedRoute && !authStore.isLoggedIn) {
    // 未登录访问受保护路由 → 重定向 /login（per D-11）
    next('/login')
    return
  }

  next()
})
```
